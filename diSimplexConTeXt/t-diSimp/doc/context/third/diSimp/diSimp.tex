% A ConTeXt document

\usemodule[t-diSimp]
\usemodule[t-literate-progs]

\showLitProgsMessages

\startLitProgsComponent diSimp

\setupinteraction[
  state=start,
  color=green,
  stype=bold
]
\placebookmarks[chapter, section, subsection][chapter, section]
\setupinteractionscreen[option=bookmark]

\setuppapersize[A4]

\setupindenting[medium,yes]

% do not indent after itemize
\setupitemgroup[itemize][indentnext=no]

% indent the paragraphs after all sectioning heads 
\setupheads[indentnext=yes]

% indent paragraphs after formulas if there is a blank space in between
\setupformulas[indentnext=auto]

\starttext

\litProgsComponent titlePages

\litProgsComponent overview

\litProgsComponent pageLayout

\litProgsComponent documentSetup

\litProgsComponent copyrights

\litProgsComponent mathEnunciations

\litProgsComponent diSimpMacros

\startbackmatter

  \litProgsComponent preamble

  \litProgsComponent conclusions
  
  \startchapter[title=Bibliography]
    \placelistofpublications
  \stopchapter
  
\stopbackmatter

\stoptext

\stopLitProgsComponent
